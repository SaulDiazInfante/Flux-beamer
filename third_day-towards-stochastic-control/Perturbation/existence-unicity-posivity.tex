            $$
        \end{yellowbox}
    \end{textblock*}

            dI(t) &=

\begin{frame}
     \frametitle{Extinction by noise}
        \begin{textblock*}{65mm}(0mm, 25mm)
            \begin{yellowbox}{Theorem}
                \begin{list}{$\bullet$}{}
                    \item
                        $
                            \displaystyle
                            \sigma ^ 2 
                            >
                            \max
                            \left\{
                                \frac{\beta}{N},
                                \frac{\beta^2}{2(\mu + \gamma)}
                            \right\}
                        $
                \end{list}
                \tcblower
                for all $I(0)\in (0, N)$ the solution to SDE($\star$)
                satisfies
                $$
                    \limsup_{t \to \infty} 
                        \frac{1}{t} \log I(t)
                        \leq
                        \underbrace{
                        -\mu - \gamma 
                        + \frac{\beta^2}{2 \sigma ^ 2}
                        }_{<0}
                $$
            \end{yellowbox}
        \end{textblock*}
\end{frame}
